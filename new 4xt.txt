    <script>
        // 1. Image Configuration
        const bgImages = [
            'https://www.eatthis.com/wp-content/uploads/sites/4/2020/07/aisle.jpg',
            'https://cdn.pixabay.com/photo/2016/03/02/20/13/grocery-1232944_1280.jpg',
            'https://cdn.pixabay.com/photo/2020/05/21/18/52/supermarket-5202138_1280.jpg',
            'https://static1.backyardbossimages.com/wordpress/wp-content/uploads/2024/04/vegetables-in-grocery-store.jpg'
        ];

        // 2. Slideshow Logic
        function startSlideshow() {
            const container = document.getElementById('slides-container');
            if (!container) return;

            bgImages.forEach((src, i) => {
                const slide = document.createElement('div');
                slide.className = `bg-slide ${i === 0 ? 'active' : ''}`;
                slide.style.backgroundImage = `url('${src}')`;
                container.appendChild(slide);
            });

            let current = 0;
            const slides = document.querySelectorAll('.bg-slide');
            
            setInterval(() => {
                slides[current].classList.remove('active');
                current = (current + 1) % slides.length;
                slides[current].classList.add('active');
            }, 6000); 
        }

        // 3. App Logic
        document.addEventListener('DOMContentLoaded', () => {
            // Initialize Background
            startSlideshow();

            // Mock Login Logic
            const authForm = document.getElementById('authForm');
            const authModal = document.getElementById('authModal');
            const dashboard = document.getElementById('dashboard');

            authForm.addEventListener('submit', (e) => {
                e.preventDefault();
                authModal.classList.add('hidden');
                dashboard.classList.remove('hidden');
            });

            document.getElementById('logoutBtn').addEventListener('click', () => {
                authModal.classList.remove('hidden');
                dashboard.classList.add('hidden');
            });
        });
    </script>
